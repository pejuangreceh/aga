Query kerusakan terbanyak
SELECT ticket.id_sub_kategori,sub_kategori.nama_sub_kategori , COUNT(ticket.id_sub_kategori) AS jumlah FROM ticket LEFT JOIN sub_kategori ON sub_kategori.id_sub_kategori = ticket.id_sub_kategori GROUP BY ticket.id_sub_kategori ORDER BY jumlah DESC

Query kantor dengan kerusakan terbanyak berdasarkan email ticket
SELECT t.email_user,u.kantor, COUNT(u.kantor) AS jumlah FROM ticket t LEFT JOIN user u ON t.email_user = u.email GROUP BY u.kantor ORDER BY u.kantor DESC
